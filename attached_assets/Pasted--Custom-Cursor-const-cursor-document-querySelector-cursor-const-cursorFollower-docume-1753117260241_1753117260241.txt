
// Custom Cursor
const cursor = document.querySelector('.cursor');
const cursorFollower = document.querySelector('.cursor-follower');

document.addEventListener('mousemove', (e) => {
  cursor.style.left = e.clientX + 'px';
  cursor.style.top = e.clientY + 'px';

  setTimeout(() => {
    cursorFollower.style.left = e.clientX + 'px';
    cursorFollower.style.top = e.clientY + 'px';
  }, 100);
});

// Hover effects for cursor
const hoverElements = document.querySelectorAll('.hover-effect');

hoverElements.forEach(el => {
  el.addEventListener('mouseenter', () => {
    cursor.style.transform = 'translate(-50%, -50%) scale(1.8)';
    cursor.style.background = 'rgba(255, 94, 14, 0.8)';
    cursorFollower.style.width = '50px';
    cursorFollower.style.height = '50px';
    cursorFollower.style.borderColor = 'rgba(255, 94, 14, 0.5)';
    cursorFollower.style.background = 'rgba(255, 94, 14, 0.1)';
  });

  el.addEventListener('mouseleave', () => {
    cursor.style.transform = 'translate(-50%, -50%)';
    cursor.style.background = 'var(--neon-orange)';
    cursorFollower.style.width = '30px';
    cursorFollower.style.height = '30px';
    cursorFollower.style.borderColor = 'var(--neon-orange)';
    cursorFollower.style.background = 'transparent';
  });
});

// Three.js for Background Scene
let scene, camera, renderer, particles;

function initBackgroundScene() {
  // Scene setup
  scene = new THREE.Scene();

  // Camera setup
  camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
  camera.position.z = 5;

  // Renderer setup
  renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
  renderer.setSize(window.innerWidth, window.innerHeight);
  renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
  document.getElementById('threejs-container').appendChild(renderer.domElement);

  // Add ambient light
  const ambientLight = new THREE.AmbientLight(0x404040, 0.5);
  scene.add(ambientLight);

  // Add directional light
  const directionalLight = new THREE.DirectionalLight(0xff5e0e, 0.8);
  directionalLight.position.set(1, 1, 1);
  scene.add(directionalLight);

  // Create floating particles
  const particlesGeometry = new THREE.BufferGeometry();
  const particlesCount = 1000;

  const posArray = new Float32Array(particlesCount * 3);

  for (let i = 0; i < particlesCount * 3; i++) {
    posArray[i] = (Math.random() - 0.5) * 10;
  }

  particlesGeometry.setAttribute('position', new THREE.BufferAttribute(posArray, 3));

  const particlesMaterial = new THREE.PointsMaterial({
    size: 0.02,
    color: 0xff5e0e,
    transparent: true,
    opacity: 0.8,
    blending: THREE.AdditiveBlending
  });

  particles = new THREE.Points(particlesGeometry, particlesMaterial);
  scene.add(particles);

  // Animation loop
  function animate() {
    requestAnimationFrame(animate);

    particles.rotation.x += 0.0005;
    particles.rotation.y += 0.0005;

    renderer.render(scene, camera);
  }

  animate();

  // Handle window resize
  window.addEventListener('resize', () => {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
  });
}

// Three.js for Product Showcase
let productScene, productCamera, productRenderer, productModel, productControls;

function initProductShowcase() {
  // Scene setup
  productScene = new THREE.Scene();
  productScene.background = null;

  // Camera setup
  productCamera = new THREE.PerspectiveCamera(75,
    document.getElementById('product-container').offsetWidth /
    document.getElementById('product-container').offsetHeight,
    0.1, 1000);
  productCamera.position.z = 3;

  // Renderer setup
  productRenderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
  productRenderer.setSize(
    document.getElementById('product-container').offsetWidth,
    document.getElementById('product-container').offsetHeight
  );
  productRenderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
  document.getElementById('product-container').appendChild(productRenderer.domElement);

  // Add lights
  const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
  productScene.add(ambientLight);

  const directionalLight1 = new THREE.DirectionalLight(0xff5e0e, 1);
  directionalLight1.position.set(1, 1, 1);
  productScene.add(directionalLight1);

  const directionalLight2 = new THREE.DirectionalLight(0xffffff, 0.8);
  directionalLight2.position.set(-1, -1, -1);
  productScene.add(directionalLight2);

  const pointLight = new THREE.PointLight(0xff5e0e, 1, 10);
  pointLight.position.set(0, 2, 2);
  productScene.add(pointLight);

  // Add orbit controls
  productControls = new THREE.OrbitControls(productCamera, productRenderer.domElement);
  productControls.enableZoom = true;
  productControls.enablePan = false;
  productControls.enableDamping = true;
  productControls.dampingFactor = 0.05;
  productControls.autoRotate = true;
  productControls.autoRotateSpeed = 0.5;
  productControls.minDistance = 2;
  productControls.maxDistance = 5;

  // Create a more sophisticated cosmetic bottle
  createPremiumBottle();

  // Animation loop
  function animateProduct() {
    requestAnimationFrame(animateProduct);
    productControls.update();
    productRenderer.render(productScene, productCamera);
  }

  animateProduct();

  // Handle window resize for product container
  window.addEventListener('resize', () => {
    productCamera.aspect =
      document.getElementById('product-container').offsetWidth /
      document.getElementById('product-container').offsetHeight;
    productCamera.updateProjectionMatrix();
    productRenderer.setSize(
      document.getElementById('product-container').offsetWidth,
      document.getElementById('product-container').offsetHeight
    );
  });
}

function createPremiumBottle() {

  const bottleGroup = new THREE.Group();

  // Bottle glass
  const bottleGeometry = new THREE.CylinderGeometry(0.5, 0.3, 1.5, 32, 1, true);
  const bottleMaterial = new THREE.MeshPhysicalMaterial({
    color: 0x111111,
    transmission: 0.95,
    roughness: 0.1,

    clearcoat: 1,
    clearcoatRoughness: 0.1,
    ior: 1.5,
    thickness: 0.3,
    specularColor: new THREE.Color(0xff5e0e),
    envMapIntensity: 1,
    emissive: 0xff5e0e,
    emissiveIntensity: 0.1

  const bottle = new THREE.Mesh(bottleGeometry, bottleMaterial);
  bottle.position.y = 0.2;
  bottleGroup.add(bottle);

  // Bottle cap
  const capGeometry = new THREE.CylinderGeometry(0.32, 0.32, 0.2, 32);
  const capMaterial = new THREE.MeshPhysicalMaterial({
    color: 0x222222,
    roughness: 0.3,
    metalness: 0.8,
    clearcoat: 1,
    clearcoatRoughness: 0.2,
    emissive: 0xff5e0e,
    emissiveIntensity: 0.05
  });

  const cap = new THREE.Mesh(capGeometry, capMaterial);
  cap.position.y = 1.0;
  bottleGroup.add(cap);

  // Liquid inside
  const liquidGeometry = new THREE.CylinderGeometry(0.48, 0.28, 1.3, 32, 1, true);
  const liquidMaterial = new THREE.MeshPhysicalMaterial({
    color: 0xff5e0e,
    transmission: 0.9,
    roughness: 0.2,
    metalness: 0.1,
    clearcoat: 1,
    clearcoatRoughness: 0.1,
    ior: 1.3,
    thickness: 0.5,
    specularIntensity: 1,
    emissive: 0xff5e0e,
    emissiveIntensity: 0.3,
    opacity: 0.8,
    transparent: true
  });

  const liquid = new THREE.Mesh(liquidGeometry, liquidMaterial);
  liquid.position.y = -0.1;
  bottleGroup.add(liquid);

  // Logo on bottle
  const logoGeometry = new THREE.PlaneGeometry(0.4, 0.2);
  const logoMaterial = new THREE.MeshBasicMaterial({
    color: 0xff5e0e,
    side: THREE.DoubleSide,
    transparent: true,
    opacity: 0.8
  });

  const logo = new THREE.Mesh(logoGeometry, logoMaterial);
  logo.position.set(0, 0.2, 0.5);
  logo.rotation.x = Math.PI / 2;
  bottleGroup.add(logo);

  productScene.add(bottleGroup);
  productModel = bottleGroup;
}
